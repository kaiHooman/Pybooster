53 def ints2ipv6(*args: int) -> str:
54  """When given eight separate integers, a hex IPv6 address is returned"""
55  if len(args) is 8:
56  for i in args:

104 def findgw() -> str:
105  """Get the Gateway IP address"""
106  with open('/proc/net/route', mode='rt', encoding='utf-8') as _file:
107  for _line in _file:
108  _field = _line.strip().split()
109  if _field[1] != '00000000' or not int(_field[3], 16) & 2:
110  continue
111  return inet_ntoa(pack('<L', int(_field[2], 16)))
