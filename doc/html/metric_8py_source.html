285 def str2tupleunit(_str: str) -> tuple:
286  """Convert measurement units from string to tuple"""
287  _num = resub('([0-9.]+)([ _,]*)([A-Za-z/]+)', r'\1', _str)
288  _unit = resub('([0-9.]+)([ _,]*)([A-Za-z/]+)', r'\3', _str)
289  return (_num, _unit)
