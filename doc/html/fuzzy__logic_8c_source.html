113 /** Minimal Negation Operator (just one false) - Return result */
114 int mno(const int count, ...) {
115  va_list varargs;
116  register int ans = 0x00, x, y, z;
117  va_start(varargs, count);
118  switch (count) {
119  case 1:

121  break;
122  case 2:
123  x = va_arg(varargs, int);
124  y = va_arg(varargs, int);
125  ans = fuzzymax(2, (fuzzymin(2, (100 - x), y)), (fuzzymin(2, x, (100 - y))));
