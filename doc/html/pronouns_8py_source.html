120 def swap_pronouns(_str: str, _pronouns: tuple) -> str:
121  """Swap pronouns"""
122  _swap = []
123  for i in _pronouns:
124  _test = re.search(i[0], _str, re.I)
125  if _test is not None:
126  _swap.append(i)
127  if not len(_swap):
128  return _str
129  else:
130  for i in _swap:
131  _str = re.sub(i[0], i[1], _str, re.I)
132  return _str
