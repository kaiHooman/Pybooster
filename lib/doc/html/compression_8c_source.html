62 /** zlib stream compression */
63 int zlib_compress_stream(FILE *source, FILE *dest, const int level) {
64  int ret, flush;
65  unsigned have;
66  z_stream strm;
67  unsigned char in[CHUNK];
68  unsigned char out[CHUNK];
69  // Allocate deflate state
70  strm.zalloc = Z_NULL;
71  strm.zfree = Z_NULL;
72  strm.opaque = Z_NULL;
73  ret = deflateInit(&strm, level);
74  if (ret != Z_OK) return ret;
75  do { // Compress until end of file
76  strm.avail_in = (unsigned int)fread(in, 1, CHUNK, source);
77  if (ferror(source)) {
78  (void)deflateEnd(&strm);
79  return Z_ERRNO;
80  }
81  flush = (feof(source) ? Z_FINISH : Z_NO_FLUSH);
82  strm.next_in = in;
83  do { // Run deflate() on input until output buffer is not full
84  strm.avail_out = CHUNK;
85  strm.next_out = out;
86  ret = deflate(&strm, flush); // No bad return value
87  if (ret == Z_STREAM_ERROR) return Z_STREAM_ERROR; // State not clobbered
88  have = (CHUNK - strm.avail_out);
89  if ((fwrite(out, sizeof(char), have, dest)) != have || ferror(dest)) {
90  (void)deflateEnd(&strm);
91  return Z_ERRNO;
92  }
93  } while (strm.avail_out == 0); // Finish compression if all data has been read
94  if (strm.avail_in != 0x00) return Z_STREAM_ERROR; // All input will be used
95  // Done when all data has been processed
96  } while (flush != Z_FINISH);
97  if (ret != Z_STREAM_END) return Z_STREAM_ERROR; // Stream is complete
98  // Clean up and return
99  (void)deflateEnd(&strm);
100  return Z_OK;
101 }

105 int zlib_compress_file(char *source_filename, char *dest_filename, const int level) {
106  FILE *source = fopen(source_filename, "rb");
107  if (source == (FILE *)NULL) {
108  fprintf(stderr, "ERROR: Failed to open file!\n%s\n", strerror(errno));
109  exit(EXIT_FAILURE);
110  }
111  FILE *dest = fopen(dest_filename, "wb");
112  if (dest == (FILE *)NULL) {
113  fprintf(stderr, "ERROR: Failed to open file!\n%s\n", strerror(errno));
114  fclose(source);
115  exit(EXIT_FAILURE);
116  }
117  // Prepare source file and buffer
118  ulint source_size = getfilesize(source_filename);
119  Bytef *source_data = (Bytef *)calloc(source_size + 1, 0x01);
120  if (source_data == (Bytef *)NULL) { ALLOC_ERROR() }
121  if (fread(source_data, 0x01, source_size, source) != source_size) {
122  free(source_data);
123  UNABLE_TO_READ_FILE()
124  }
125  fclose(source);
126  source_data[source_size + 1] = '\0';
127  // Prepare dest file
128  ulint dest_size = compressBound(source_size + 1024);
129  Bytef *dest_buf = (Bytef *)calloc(dest_size, 0x01);
130  if (dest_buf == (Bytef *)NULL) {
131  fclose(dest);
132  free(source_data);
133  free(dest_buf);
134  ALLOC_WRITE_ERROR()
135  }
136  // Compress data
137  int ret;
138  ret = compress2(dest_buf, &dest_size, source_data, source_size, level);
139  if (ret != Z_OK) {
140  fclose(dest);
141  free(dest_buf);
142  return ret;
143  }
144  // Write compressed data
145  setbuf(dest, (char *)dest_buf);
146  if ((fwrite((char *)dest_buf, 0x01, dest_size, dest)) != dest_size) {
147  fclose(dest);
148  free(dest_buf);
149  UNABLE_TO_WRITE_FILE()
150  }
151  // Clean up and return
152  fclose(dest);
153  free(dest_buf);

158 /** zlib stream decompression */
159 int zlib_decompress_stream(FILE *source, FILE *dest) {
160  int ret;
161  unsigned have;
162  z_stream strm;
163  unsigned char in[CHUNK];
164  unsigned char out[CHUNK];
165  // Allocate inflate state
166  strm.zalloc = Z_NULL;
167  strm.zfree = Z_NULL;
168  strm.opaque = Z_NULL;
169  strm.avail_in = 0;
170  strm.next_in = Z_NULL;
171  ret = inflateInit(&strm);
172  if (ret != Z_OK) return ret;
173  do { // Decompress until deflate stream ends or end of file
174  strm.avail_in = (unsigned int)fread(in, 1, CHUNK, source);
175  if (ferror(source)) {
176  (void)inflateEnd(&strm);
177  return Z_ERRNO;
178  }
179  if (strm.avail_in == 0x00) break;
180  strm.next_in = in;
181  do { // Run inflate() on data until the output buffer is not full
182  strm.avail_out = CHUNK;
183  strm.next_out = out;
184  ret = inflate(&strm, Z_NO_FLUSH);
185  if (ret == Z_STREAM_ERROR) return Z_STREAM_ERROR; // State not clobbered
186  switch (ret) {
187  case Z_NEED_DICT:
188  ret = Z_DATA_ERROR; // Fall through
189  case Z_DATA_ERROR:
190  case Z_MEM_ERROR:
191  (void)inflateEnd(&strm);
192  return ret;
193  default: continue;
194  }
195  have = (CHUNK - strm.avail_out);
196  if ((fwrite(out, sizeof(char), have, dest)) != have || ferror(dest)) {
197  (void)inflateEnd(&strm);
198  return Z_ERRNO;
199  }
200  } while (strm.avail_out == 0);
201  } while (ret != Z_STREAM_END); // Done when inflate() finishes
202  // Clean up and return
203  (void)inflateEnd(&strm);
204  return ((ret == Z_STREAM_END) ? Z_OK : Z_DATA_ERROR);
205 }

208 /** zlib file decompression */
209 int zlib_decompress_file(char *source_filename, char *dest_filename) {
210  FILE *source = fopen(source_filename, "rb");
211  if (source == NULL) {
212  fprintf(stderr, "ERROR: Failed to open source file!\n%s\n", strerror(errno));
213  exit(EXIT_FAILURE);
214  }
215  FILE *dest = fopen(dest_filename, "wb");
216  if (dest == NULL) {
217  fprintf(stderr, "ERROR: Failed to open file!\n%s\n", strerror(errno));
218  fclose(source);
219  exit(EXIT_FAILURE);
220  }
221  // Prepare source file and buffer
222  ulint source_size = getfilesize(source_filename);
223  uint8_t *source_data = (uint8_t *)calloc(source_size + 1, 0x01);
224  if (source_data == NULL) { ALLOC_ERROR() }
225  if (fread(source_data, 0x01, source_size + 1, source) != source_size) {
226  fputs("`zlib_decompress_file` (Reading source file): ", stderr);
227  free(source_data);
228  fclose(source);
229  fclose(dest);
230  UNABLE_TO_READ_FILE()
231  }
232  fclose(source);
233  source_data[source_size + 1] = '\0';
234  // Allocate inflate state
235  unsigned char output_data[CHUNK] = { 0 };
236  z_stream strm;
237  strm.zalloc = Z_NULL;
238  strm.zfree = Z_NULL;
239  strm.opaque = Z_NULL;
240  strm.avail_in = (unsigned int)source_size;
241  strm.next_in = (Bytef *)source_data;
242  strm.data_type = Z_BINARY;
243  int ret = inflateInit(&strm);
244  if (ret != Z_OK) {
245  free(source_data);
246  fclose(dest);
247  return ret;
248  }
249  size_t dest_size = 0x00;
250  // Decompress data
251  do { // Run inflate() on data until finished
252  if (strm.avail_in == 0) break;
253  strm.avail_out = CHUNK;
254  strm.next_out = (Bytef *)output_data;
255  ret = inflate(&strm, Z_NO_FLUSH);
256  if (ret == Z_STREAM_ERROR) { // State not clobbered
257  free(source_data);
258  fclose(dest);
259  return Z_STREAM_ERROR;
260  }
261  switch (ret) {
262  case Z_NEED_DICT:
263  ret = Z_DATA_ERROR; // Fall through
264  case Z_DATA_ERROR:
265  case Z_MEM_ERROR:
266  (void)inflateEnd(&strm);
267  free(source_data);
268  fclose(dest);
269  return ret;
270  default:
271  dest_size += (size_t)(CHUNK - strm.avail_out);
272  if (strm.avail_in == 0 || strm.avail_out == 0 || ret == Z_STREAM_END) break;
273  }
274  } while (ret != Z_STREAM_END && strm.avail_out == 0); // Wait until inflate() finishes
275  (void)inflateEnd(&strm);
276  Bytef *dest_buf = (Bytef *)calloc(dest_size, 0x01);
277  ret = uncompress(dest_buf, &dest_size, source_data, source_size);
278  if (ret != Z_OK) {
279  fclose(dest);
280  free(dest_buf);
281  free(source_data);
282  return ret;
283  }
284  free(source_data);
285  // Write compressed data
286  setbuf(dest, NULL);
287  if ((fwrite((char *)dest_buf, 0x01, dest_size, dest)) != dest_size) {
288  fclose(dest);
289  free(dest_buf);
290  fputs("`zlib_decompress_file` (Writing compressed file): ", stderr);
291  UNABLE_TO_WRITE_FILE()
292  }
293  // Clean up and return
294  fclose(dest);
295  free(dest_buf);
296  return Z_OK;
297 }

321  fputs("Unknown error\n", stderr);
