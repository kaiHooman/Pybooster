77 /** Generate a tone lasting one second; remember to use `free(sound.buffer);` */
78 tone generatetone(const float freq, ao_sample_format format) {
79  tone sound;
80  sound.buf_size = (format.bits >> 3) * format.channels * format.rate;
81  register int i;
82  register float _val;
83  for (i = 0; i < format.rate; ++i) {
84  _val = (((float)i / (float)format.rate) * (float)6.283185F);
85  sound.sample = (int)((float)24576.0F * sinf(freq * _val)); // amplitude * sin_wave
86  sound.buffer[4 * i] = sound.buffer[4 * i + 2] = (char)(sound.sample & 0xff);
87  sound.buffer[4 * i + 1] = sound.buffer[4 * i + 3] = (char)((sound.sample >> 0x08) & 0xff);
88  }
89  return sound;
90 }
